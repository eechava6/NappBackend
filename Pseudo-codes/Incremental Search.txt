read x0, step, iteration-number
x0 <- f(x0)
if fx0 = 0 then
	x0 is root
else
	x1 <- x0 + step
	count <- 1
	fx1 <- f(x1)
	while fx0 * fx1 > 0 and count < iteration-number do
		x0 <- x1
		fx0 <- fx1
		x1 <- x0 + step
		fx1 <- f(x1)
		count <- count + 1
	end while
	if fx1 = 0 then
		x1 is root
	else if fx0 * fx1 < 0 then
		there is a root between x0 and x1
	else
		failed in iteration-number iterations
	endif
endif
