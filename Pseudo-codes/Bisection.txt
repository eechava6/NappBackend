read xi, xs, toleration, iteration-number
fxi <- f(xi)
fxs <- f(xs)
if fxi = 0 then
	xi is root
else if fxs = 0 then
	xs is root
else if fxi * fxs < 0 then
	xm <- (xi + xs)/2
	fxm <- f(xm)
	count <- 1
	error <- toleration + 1
	while error > toleration and fxm not equal 0 and count < iteration-number do
		if fxi * fxm < 0 then
			xs <- xm
			fxs <- fxm
		else
			xi <- xm
			fxi <- fxm
		end if
		xaux <- xm
		xm <- (xi + xs)/2
		fxm <- f(xm)
		error <- abs(xm - xaux)
		count <- count + 1
	end while
	if fxm = 0 then
		xm is root
	else if error < toleration then
		xm is approximation to a root with a toleration <- toleration
	else
		failed in iteration-number iterations
	endif
else
	the interval is not adequate
endif
