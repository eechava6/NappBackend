read tolerance, x0, iteration-number
fx <- f(x0)
dfx <- f'(x0)
count <- 0
error <- tolerance + 1
while error > tolerance and count < iteration-number and fx != 0 and dfx != 0
	xn = x0 - (fx / dfx)
	fx = f(xn)
	dfx = df(xn)

	error = |xn -x0|
	x0 = xn
	count = count + 1
if fx == 0
	x0 is a root
else if error <= tolerance
	x0 is an approximation with tolerance tolerance
else if dfx == 0
	xn is a possible multiple root
else
	maximum iterations
