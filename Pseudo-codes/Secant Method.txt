read x0, x1, iteration-number, tolerance
fx0 = f(x0)
if fx0 = 0 then
	x0 is root
else
	fx1 = f(x1)
	count = 0
	error = tolerance + 1
	denominator = fx1 - fx0
	while error > tolerance and fx1 != 0 and denominator != 0 and count < iteration-number do
		x2 = x1 - fx1 * (x1 - x0)/denominator
		error = |x2 - x1|
		x0 = x1
		fx0 = fx1
		x1 = x2
		fx1 = f(x1)
		denominator = fx1 - fx0
		count = count + 1
	end while
	if fx1 = 0 then
		x1 is root
	else if error < tolerance then
		x1 is an approximation to a root with a tolerance tolerance
	else if denominator = 0 then 
		there's a possible multiple root
	else
		failed in iteration-number iterations
	endif
endif
