read xi, tolerance, iteration-number
fxi = f(xi)
dfxi = df(xi)
ddfxi = ddf(xi)
count = 1
error = tolerance + 1
while error > tolerance and count <= iteration-number
	xitmp = xi
	xi = xn(xi)
	fxi = f(xi)
	dfxi = df(xi)
	ddfxi = ddf(xi)
	error = |xi - xitmp|
	if fxi == 0 then
		root found
	else if error < tolerance then
		error lower than tolerance
	else if count >= iteration-number then
		maximum iterations
	count = count + 1
function xn(xn)
	xn - ((f(xn)*df(xn))/((df(xn)**2)-f(xn)*ddf(xn)))
